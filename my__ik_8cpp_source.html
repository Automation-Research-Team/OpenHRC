1076  // std::cout << a_norm << " " << b_norm << std::endl;

1079  Vector3d nb = b / b_norm;

1085  double d0 = na.dot(b0 - a0);
1086  double d1 = nb.dot(b1 - a0);

1089  if (std::abs(d0) < std::abs(d1)) {
1090  as = 0.0;
1091  bs = 0.0;
1092  return true;
1093  } else if (d0 > a_norm && d1 > a_norm) {
1094  if (std::abs(d0) < std::abs(d1)) {
1095  as = 1.0;
1096  bs = 0.0;
1097  return true;
1098  } else {
1099  as = 1.0;
1100  bs = 1.0;
1101  return true;
1102  }
1103  }
1104  }

1106  as = 0.0;
1107  bs = 0.0;
1108  return true;
1109

1117  if (as < 0.0) {
1118  as = 0.0;

1120  as = 1.0;
1121  }
1122

1125  if (dot < 0)
1126  bs = 0.0;
1127  else if (dot > b_norm)

1129  else
